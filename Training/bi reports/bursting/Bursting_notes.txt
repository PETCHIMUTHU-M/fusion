'RTF' TEMPLATE_FORMAT,
'EN-US' LOCALE,
'PDF' OUTPUT_FORMAT,
'EMAIL' DEL_CHANNEL,
'ROOPA.RAMACHANDRAN@CHAINSYS.COM'PARAMETER1,
'DONOTREPLY@CHAINSYS.COM' PARAMETER2,
'HARINIROOPA5@GMAIL.COM' PARAMETER3,
'REG PRACTICE BURSTING' PARAMETER4,
'HAI MAIL FROM ROOPA CHAINSYS' PARAMETER5,
'TRUE'  PARAMETER6

FROM 
RA_CUSTOMER_TRX_ALL RC,HR_ALL_ORGANIZATION_UNITS HRO,
HZ_CUST_ACCOUNTS HZA,HZ_PARTIES HZP

WHERE 1=1
AND HRO.ORGANIZATION_ID=RC.ORG_ID
AND RC.BILL_TO_CUSTOMER_ID=HZA.CUST_ACCOUNT_ID
AND HZP.PARTY_ID=HZA.PARTY_ID
AND TRX_NUMBER IN ('18','1029')
